CREATE TABLE SERVER(
    ID BIGINT NOT NULL PRIMARY KEY,
    NAME VARCHAR(100),

    ADMIN_ROLE BIGINT,
    SUS_ROLE BIGINT,

    WELCOME_CHANNEL BIGINT,
    DAILY_CHANNEL BIGINT,
    LOG_CHANNEL BIGINT,

    SLUR_FILTER BOOLEAN,
    CHAT_WORDS_PERCENTAGE INTEGER,
    PREFIX VARCHAR(1)
);

CREATE TABLE SERVER_BAN_LIST(
    ID BIGINT NOT NULL PRIMARY KEY,
    SERVER_ID BIGINT,
    NAME VARCHAR(32),
    ENFORCER BIGINT,
    REASON VARCHAR(100),
    BAN_DATE DATE,
    BAN_DURATION DATE,
    FOREIGN KEY (SERVER_ID) REFERENCES SERVER(ID)
);

CREATE TABLE OPEN_BAN_LIST(
    ID BIGINT NOT NULL PRIMARY KEY,
    NAME VARCHAR(32),
    ENFORCER BIGINT,
    REASON VARCHAR(100),
    BAN_DATE DATE,
    BAN_DURATION DATE
);

CREATE TABLE LEVELS(
    ID INTEGER NOT NULL PRIMARY KEY,
    POINTS BIGINT,
    LEVEL BIGINT
);

CREATE TABLE MEDIA(
    ID INTEGER NOT NULL PRIMARY KEY,
    TOTAL_UPVOTES BIGINT,
    TOTAL_DOWNVOTES BIGINT,
    TOTAL_REPORTS BIGINT,
    IS_SAVE BOOLEAN
);
