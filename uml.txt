CREATE TABLE SERVER(
    ID INTEGER NOT NULL PRIMARY KEY,
    NAME VARCHAR(100),

    MOD_ROLE INTEGER,
    SUS_ROLE INTEGER,

    WELCOME_CHANNEL INTEGER,
    DAILY_CHANNEL INTEGER,
    LOG_CHANNEL INTEGER,

    SLUR_FILTER BOOLEAN,
    CHAT_WORDS_PERCENTAGE INTEGER,
    PREFIX VARCHAR(1)
);

CREATE TABLE SERVER_BAN_LIST(
    ID INTEGER NOT NULL PRIMARY KEY,
    SERVER_ID INTEGER,
    NAME VARCHAR(32),
    ENFORCER INTEGER,
    REASON VARCHAR(100),
    BAN_DATE DATE,
    BAN_DURATION DATE,
    FOREIGN KEY (SERVER_ID) REFERENCES SERVER(ID)
);

CREATE TABLE OPEN_BAN_LIST(
    ID INTEGER NOT NULL PRIMARY KEY,
    NAME VARCHAR(32),
    ENFORCER INTEGER,
    REASON VARCHAR(100),
    BAN_DATE DATE,
    BAN_DURATION DATE
);
